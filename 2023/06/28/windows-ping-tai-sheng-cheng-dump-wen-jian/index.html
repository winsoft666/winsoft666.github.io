	TCHAR *lstrrchr(LPCTSTR string, int ch) {
		TCHAR *start = (TCHAR *)string;
 
		while (*string++)
			;
 
		while (--string != start && *string != (TCHAR)ch)
			;
 
		if (*string == (TCHAR)ch)
			return (TCHAR *)string;
 
		return NULL;
	}
